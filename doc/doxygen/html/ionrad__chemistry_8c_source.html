3  * FILE: ionrad_chemistry.c
4  *
5  * PURPOSE: Contains functions to compute various chemistry updates
6  * used in the Krumholz, Stone, & Gardiner (2007) ionizing radiation
7  * transfer algorithm. Unless otherwise specified, an input of T is
8  * temperature in Kelvin and x is ion fraction (H+ / (H+ + H).
9  *
10  * Use of these routines requires that --enable-ion-radiation be set
11  * at compile time.
12  *
13  * CONTAINS PUBLIC FUNCTIONS:
14  * recomb_rate_coef(T): e- + H+ -> H rate coefficient in cm^3 s^-1
15  * coll_ion_rate_coef(T): collisional H ionization rate coefficient in
16  * cm^3 s^-1
17  * recomb_cool_rate_coef(T): radiative recombination cooling rate
18  * coefficient in erg cm^3 s^-1
19  * dmc_cool_rate(x, T): Dalgarno-McCray (1973) atomic gas cooling rate
20  * in erg cm^3 s^-1
21  * ki_cooling_rate(x, T): Koyama & Inutsuka (2002) cooling rate, including
22  * H_2 and CO cooling, in erg cm^3 s^-1. Only applies to molecular
23  * gas.
24  * ki_heating_rate(x, T): Koyama & Inutsuka (2002) heating rate, in erg
25  * cm^-3 s^-1. Only applies to non-ionized gas.
26  * osterbrock_cool_rate(T): cooling rate computed using transitions
27  * of first two ionized states of O, N, and Ne, as taken from
28  * Osterbrock & Ferland 2006, in erg cm^3 s^-1
29  *
30  * Added 03/13/14 by A. Tripathi:
31  * lya_cool_rate(nh, ne, T): Black (1981) Lya cooling rate, in erg cm^-3 s^-1
32  */

153 Real dmc_cool_rate(Real x, Real T) {
154 #ifndef NORECOMBINATION
155  /* Equilibrium cooling rate from Dalgarno & McCray (1972) */
156  /* Implementation follows Jim Stone's dmc routine */
157  Real le, lh, u, u2, om, dom, p1;
158  Real qq2, qt1, qt2, qt3, qt4, xu1, xu2, xu3, xu4, tlost, tcool;
159  int ipps, jaug;
160 
161  if (x < 1.0e-3) x = 1.0e-3;
162 
163  /* Electron impact excitation luminosity (eqn 3-10) */
164  le = 0.0;
165  if (T > 10) {
166  le += 2.96e-23/sqrt(T) * exp(-92.0/T);
167  if (T > 50) {
168  le += 6.08e-23/sqrt(T) * exp(-413.0/T)
169  + 3.52e-23/sqrt(T) *
170  (exp(-554.0/T) + 1.3* exp(-961.0/T));
171  if (T > 2.0e4) {
172  le = le + 4.14e-26*sqrt(T) * exp(-22700.0/T)
173  + 7.13e-26*sqrt(T)*(1.0-2.7e-9*T*T)
174  *exp(-27700.0/T);
175  }
176  }
177  }
178 
179  /* Hydrogen cooling */
180  if (T > 50.0) {
181  lh = 2.37e-27*exp(-413/T)
182  + 3.52e-27*(exp(-554.0/T)+1.4*exp(-961.0/T));
183  } else {
184  lh = 0.0;
185  }
186 
187  /* Neutral cooling */
188  u = T/157890. < 3.16 ? T/157890. : 3.16;
189  u2 = u*u;
190  om=.6098+1.489*u+.50755*u2-.38145*u*u2+.10196*u2*u2
191  -.01007*u*u2*u2;
192  dom=(1.489+2.*.50755*u-3.*.38145*u2+4.*.10196*u2*u
193  -5.*.01007*u2*u2)/157890.;
194  p1 = 0.0;
195  if (T > 1.0e4)
196  p1 = 0.5*1.41e-16*om*exp(-118000./T) / sqrt(T);
197 
198  /* Cooling in various regimes: < 100 K */
199  if (T < 100.0)
200  /* return(0.0); */
201  return(x*le + lh +
202  (1.0-x)*p1);
203  if (T < 1.0e4)
204  return(SCALEFACTOR*x*2.8347e-10*pow(T - 1.0e+02, 2.3562)
205  + x*le + lh
206  +(1.0-x)*p1);
207  if (T > 1.27717e8)
208  return(x*2.3988e-04*sqrt(T));
209 
210  tlost = log(T)/log(10.0);
211  ipps = (int) floor(10.0*tlost) - 38;
212  if (ipps > 41) ipps = 41;
213  jaug = 2 > ipps ? 2 : ipps;
214  qq2 = 3.8 + 0.1*jaug;
215  qt2 = tlost - qq2;
216  qt3 = qt2 - 0.1;
217 
218  if ((jaug == 2) || (jaug == 41)) {
219  tcool = (xmat[jaug-1]*qt2 - xmat[jaug-2]*qt3)*10.0;
220  return(SCALEFACTOR*pow(10.0, tcool)*x + (1.0-x)*p1);
221  }
222 
223  qt1 = qt2 + 0.1;
224  qt4 = qt3 - 0.1;
225 
226  xu1 = qt2*qt3*qt4/6.0e-03;
227  xu2 = qt1*qt3*qt4/2.0e-03;
228  xu3 = qt1*qt2*qt4/2.0e-03;
229  xu4 = qt1*qt2*qt3/6.0e-03;
230 
231  tcool = -xmat[jaug-3]*xu1 + xmat[jaug-2]*xu2 -
232  xmat[jaug-1]*xu3 + xmat[jaug]*xu4;
233  return(SCALEFACTOR*pow(10.0,tcool)*x + (1.0-x)*p1);
234 #else
235  return(0.0);
236 #endif
237 }
238 #undef SCALEFACTOR
