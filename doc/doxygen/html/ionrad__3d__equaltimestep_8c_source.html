286 #define DAMPFACTOR 0.5
287 Real compute_chem_rates(GridS *pGrid)
288 {
289  int i, j, k, n;
290  Real n_H, n_Hplus, n_e, d_nlim;
291  Real e_sp, T, x;
292  Real dt_chem, dt_chem1, dt_chem2, dt_chem_min;
293 #ifdef MPI_PARALLEL
294  int err;
295  Real dt_chem_min_glob;
296 #endif
297 
298  /* Initialize chemistry time step to large time step */
299  dt_chem_min = LARGE;
300 
301  /* Loop over cells to get timestep */
302  for (k=pGrid->ks; k<=pGrid->ke; k++) {
303  for (j=pGrid->js; j<=pGrid->je; j++) {
304  for (i=pGrid->is; i<=pGrid->ie; i++) {
305 
306  /* Get species abundances */
307  n_H = pGrid->U[k][j][i].s[0] / m_H;
308  n_Hplus = (pGrid->U[k][j][i].d - pGrid->U[k][j][i].s[0]) / m_H;
309  n_e = n_Hplus + pGrid->U[k][j][i].d * alpha_C / (14.0 * m_H);
310  x = n_e / (n_H + n_Hplus);
311 
312  /* Get gas temperature in K */
313  e_sp = (pGrid->U[k][j][i].E -
314  0.5 * (pGrid->U[k][j][i].M1*pGrid->U[k][j][i].M1 +
315  pGrid->U[k][j][i].M2*pGrid->U[k][j][i].M2 +
316  pGrid->U[k][j][i].M3*pGrid->U[k][j][i].M3)
317  / pGrid->U[k][j][i].d
318 #ifdef MHD
319  - 0.5 * (pGrid->U[k][j][i].B1c*pGrid->U[k][j][i].B1c +
320  pGrid->U[k][j][i].B2c*pGrid->U[k][j][i].B2c +
321  pGrid->U[k][j][i].B3c*pGrid->U[k][j][i].B3c)
322 #endif
323  ) / pGrid->U[k][j][i].d;
324  T = Gamma_1 * e_sp * (x*0.5*m_H+(1.0-x)*mu)/ k_B;
325  if (T < tfloor) T = tfloor;
326 
327  /* Get rate of change of neutral density */
328  nHdot[k][j][i] =
329  recomb_rate_coef(T) * time_unit * n_e * n_Hplus
330  - ph_rate[k][j][i] * n_H
331  - coll_ion_rate_coef(T) * time_unit * n_e * n_H;
332 
333  /* Check if the sign has flipped -- oscillatory overstability
334  check */
335  if (nHdot[k][j][i] < 0.0) {
336  if (last_sign[k][j][i] == 1) sign_count[k][j][i]++;
337  else if (sign_count[k][j][i] > 0) sign_count[k][j][i]--;
338  last_sign[k][j][i] = -1;
339  } else if (nHdot[k][j][i] > 0.0) {
340  if (last_sign[k][j][i] == -1) sign_count[k][j][i]++;
341  else if (sign_count[k][j][i] > 0) sign_count[k][j][i]--;
342  last_sign[k][j][i] = 1;
343  } else {
344  sign_count[k][j][i] = last_sign[k][j][i] = 0;
345  }
346 
347  /* If sign has flipped too many times successively, this cell
348  * is probably experiencing oscillatory overstability. To
349  * combat this, decrease the update amount by a chosen factor
350  * for every repeat of the same sign past the trigger number
351  */
352  for (n=MAXSIGNCOUNT; n<sign_count[k][j][i]; n++) {
353  edot[k][j][i] *= DAMPFACTOR;
354  nHdot[k][j][i] *= DAMPFACTOR;
355  }
356 
357  /* Get ionization fraction floor */
358  d_nlim = pGrid->U[k][j][i].d*IONFRACFLOOR;
359  d_nlim = d_nlim < d_nlo ? d_nlim : d_nlo;
360 
361  /* Compute chemistry time step for this cell and find the min */
362  if (nHdot[k][j][i] == 0.0) {
363  dt_chem1 = dt_chem2 = LARGE;
364  } else if (nHdot[k][j][i] > 0.0) {
365  dt_chem1 = max_dx_iter / (1+max_dx_iter) * n_e / nHdot[k][j][i];
366  dt_chem2 = max_dx_iter * n_H / nHdot[k][j][i];
367  } else if (pGrid->U[k][j][i].s[0] > 1.0001*d_nlim) {
368  dt_chem1 = -max_dx_iter * n_e / nHdot[k][j][i];
369  dt_chem2 = -max_dx_iter / (1+max_dx_iter) * n_H / nHdot[k][j][i];
370  } else {
371  dt_chem1 = dt_chem2 = LARGE;
372  }
373  dt_chem = (dt_chem1 < dt_chem2) ? dt_chem1 : dt_chem2;
374  dt_chem_min = (dt_chem < dt_chem_min) ? dt_chem : dt_chem_min;
375 
376  if (dt_chem < 0) {
377  ath_error("[compute_chem_rates]: cell %d %d %d: dt_chem = %e, d = %e, d_n = %e, T = %e, nH = %e, nH+ = %e, ne = %e, nHdot = %e\n", i,j,k, dt_chem, pGrid->U[k][j][i].d, pGrid->U[k][j][i].s[0], T, n_H, n_Hplus, n_e, nHdot[k][j][i]);
378  }
379  }
380  }
381  }
382 
383 #ifdef MPI_PARALLEL
384  /* Sync chemistry timestep across processors */
385  err = MPI_Allreduce(&dt_chem_min, &dt_chem_min_glob, 1, MP_RL,
386  MPI_MIN, MPI_COMM_WORLD);
387  if(err) ath_error("[compute_chem_rates]: MPI_Allreduce returned error code %d\n"
388  ,err);
389  dt_chem_min = dt_chem_min_glob;
390 #endif
391 
392  /*fclose(fp);*/
393 
394  return(dt_chem_min);
395 }

400 Real compute_therm_rates(GridS *pGrid)
401 {
402  int i, j, k;
403  Real n_H, n_Hplus, n_e, e_thermal;
404  Real e_sp, T, x, e_sp_min, e_th_min, e_min, d_nlim;
405  Real dt_therm, dt_therm1, dt_therm2, dt_therm_min;
406 #ifdef MPI_PARALLEL
407  int err;
408  Real dt_therm_min_glob;
409 #endif
410 
411  /* Initialize thermal time step to large value */
412  dt_therm_min = LARGE;
413 
414  /* Loop over cells to get timestep */
415  for (k=pGrid->ks; k<=pGrid->ke; k++) {
416  for (j=pGrid->js; j<=pGrid->je; j++) {
417  for (i=pGrid->is; i<=pGrid->ie; i++) {
418 
419  /* Get species abundances */
420  n_H = pGrid->U[k][j][i].s[0] / m_H;
421  n_Hplus = (pGrid->U[k][j][i].d - pGrid->U[k][j][i].s[0]) / m_H;
422  n_e = n_Hplus + pGrid->U[k][j][i].d * alpha_C / (14.0 * m_H);
423  x = n_e / (n_H + n_Hplus);
424 
425  /* Get gas temperature in K */
426  e_thermal = pGrid->U[k][j][i].E - 0.5 *
427  (pGrid->U[k][j][i].M1 * pGrid->U[k][j][i].M1 +
428  pGrid->U[k][j][i].M2 * pGrid->U[k][j][i].M2 +
429  pGrid->U[k][j][i].M3 * pGrid->U[k][j][i].M3)
430  / pGrid->U[k][j][i].d
431 #ifdef MHD
432  - 0.5 * (pGrid->U[k][j][i].B1c * pGrid->U[k][j][i].B1c +
433  pGrid->U[k][j][i].B2c * pGrid->U[k][j][i].B2c +
434  pGrid->U[k][j][i].B3c * pGrid->U[k][j][i].B3c)
435 #endif
436  ;
437  e_sp = e_thermal / pGrid->U[k][j][i].d;
438  T = Gamma_1 * e_sp * (x*0.5*m_H+(1.0-x)*mu)/ k_B;
439 
440  /* Check temperature floor. If cell is below temperature
441  floor, skip it. We'll fix it later */
442  if (T < tfloor) {
443  edot[k][j][i] = 0.0;
444  continue;
445  }
446 
447  /* If we're at the ionization floor and trying to ionize
448  further, we won't update the temperature, so skip this
449  cell. */
450  d_nlim = pGrid->U[k][j][i].d*IONFRACFLOOR;
451  d_nlim = d_nlim < d_nlo ? d_nlim : d_nlo;
452  if ((nHdot[k][j][i] < 0) &&
453  (pGrid->U[k][j][i].s[0] < 1.0001*d_nlim)) {
454  edot[k][j][i] = 0.0;
455  continue;
456  }
457 
458  /* Get rate of change of gas energy. Only use molecular cooling
459  in cells with < COOLFRAC or > 1 - COOLFRAC ionization fraction, to
460  avoid artificial over-cooling in mixed or transition cells. */
461  edot[k][j][i] = ph_rate[k][j][i] * e_gamma * n_H
462  - osterbrock_cool_rate(T) * n_e*n_Hplus
463  + recomb_cool_rate_coef(T) * time_unit * n_Hplus * n_e;
464  if ((n_Hplus / (n_H+n_Hplus) < COOLFRAC) ||
465  (n_Hplus / (n_H+n_Hplus) > 1.0-COOLFRAC)) {
466  edot[k][j][i] += ki_heat_rate() * time_unit * n_H
467  - ki_cool_rate(T) * time_unit * n_H * n_H;
468  }
469 
470  /* Compute thermal time step for this cell and find the
471  min. Note that, if we're cooling, we need to take into
472  account the effect of the floor. */
473  if (edot[k][j][i] == 0.0) {
474  dt_therm1 = dt_therm2 = LARGE;
475  } else if (edot[k][j][i] > 0.0) {
476 
477  /* We're heating, no need to consider floor */
478  dt_therm1 = max_de_iter * pGrid->U[k][j][i].E / edot[k][j][i];
479  dt_therm2 = max_de_therm_iter * e_thermal / edot[k][j][i];
480 
481  } else {
482 
483  /* We're cooling. Start by computing the total and thermal
484  energy the gas would have if it were at the temperature
485  floor. */
486  e_sp_min = tfloor * k_B / ((x*0.5*m_H+(1.0-x)*mu) * Gamma_1);
487  e_th_min = e_sp_min * pGrid->U[k][j][i].d;
488  e_min =
489  0.5 * (pGrid->U[k][j][i].M1*pGrid->U[k][j][i].M1 +
490  pGrid->U[k][j][i].M2*pGrid->U[k][j][i].M2 +
491  pGrid->U[k][j][i].M3*pGrid->U[k][j][i].M3)
492  / pGrid->U[k][j][i].d
493 #ifdef MHD
494  + 0.5 * (pGrid->U[k][j][i].B1c*pGrid->U[k][j][i].B1c +
495  pGrid->U[k][j][i].B2c*pGrid->U[k][j][i].B2c +
496  pGrid->U[k][j][i].B3c*pGrid->U[k][j][i].B3c)
497 #endif
498  + e_th_min;
499 
500  /* If cooling to the temperature floor would not violate the
501  constraint on the maximum allowable change in either
502  total or thermal energy, there is no constraint, and we
503  can continue to the next cell */
504  if ((e_thermal/(1.0+max_de_therm_iter) < e_th_min) &&
505  (pGrid->U[k][j][i].E/(1.0+max_de_iter) < e_min))
506  continue;
507 
508  /* If we're here, cooling to the temperature floor would
509  violate our time step constraint. Therefore compute the
510  time step normally. */
511  dt_therm1 = -max_de_iter / (1+max_de_iter) * pGrid->U[k][j][i].E
512  / edot[k][j][i];
513  dt_therm2 = -max_de_therm_iter / (1+max_de_therm_iter) * e_thermal /
514  edot[k][j][i];
515 
516  }
517 
518  /* Set time step to minimum */
519  dt_therm = (dt_therm1 < dt_therm2) ? dt_therm1 : dt_therm2;
520  dt_therm_min = (dt_therm < dt_therm_min) ? dt_therm : dt_therm_min;
521  }
522  }
523  }
524 
525 #ifdef MPI_PARALLEL
526  /* Sync thermal timestep across processors */
527  err = MPI_Allreduce(&dt_therm_min, &dt_therm_min_glob, 1, MP_RL,
528  MPI_MIN, MPI_COMM_WORLD);
529  if(err) ath_error("[compute_therm_rates]: MPI_Allreduce returned error code %d\n"
530  ,err);
531  dt_therm_min = dt_therm_min_glob;
532 #endif
533 
534  return(dt_therm_min);

566 Real compute_dt_hydro(GridS *pGrid) {
567  int i,j,k;
568  Real di,v1,v2,v3,qsq,p,asq,cf1sq,cf2sq,cf3sq,max_dti=0.0,dt;
569 #ifdef MHD
570  Real b1,b2,b3,bsq,tsum,tdif;
571 #endif /* MHD */
572 #ifdef MPI_PARALLEL
573  Real dt_glob;
574  int err;
575 #endif /* MPI_PARALLEL */
576 
577  for (k=pGrid->ks; k<=pGrid->ke; k++) {
578  for (j=pGrid->js; j<=pGrid->je; j++) {
579  for (i=pGrid->is; i<=pGrid->ie; i++) {
580  di = 1.0/(pGrid->U[k][j][i].d);
581  v1 = pGrid->U[k][j][i].M1*di;
582  v2 = pGrid->U[k][j][i].M2*di;
583  v3 = pGrid->U[k][j][i].M3*di;
584  qsq = v1*v1 + v2*v2 + v3*v3;
585 
586 #ifdef MHD
587  /* Use maximum of face-centered fields (always larger than
588  cell-centered B) */
589  b1 = pGrid->U[k][j][i].B1c
590  + fabs((double)(pGrid->B1i[k][j][i] - pGrid->U[k][j][i].B1c));
591  b2 = pGrid->U[k][j][i].B2c
592  + fabs((double)(pGrid->B2i[k][j][i] - pGrid->U[k][j][i].B2c));
593  b3 = pGrid->U[k][j][i].B3c
594  + fabs((double)(pGrid->B3i[k][j][i] - pGrid->U[k][j][i].B3c));
595  bsq = b1*b1 + b2*b2 + b3*b3;
596  /* compute sound speed squared */
597 #ifdef ADIABATIC
598  p = MAX(Gamma_1*(pGrid->U[k][j][i].E - 0.5*pGrid->U[k][j][i].d*qsq
599  - 0.5*bsq), TINY_NUMBER);
600  asq = Gamma*p*di;
601 #else
602  asq = Iso_csound2;
603 #endif /* ADIABATIC */
604  /* compute fast magnetosonic speed squared in each direction */
605  tsum = bsq*di + asq;
606  tdif = bsq*di - asq;
607  cf1sq = 0.5*(tsum + sqrt(tdif*tdif + 4.0*asq*(b2*b2+b3*b3)*di));
608  cf2sq = 0.5*(tsum + sqrt(tdif*tdif + 4.0*asq*(b1*b1+b3*b3)*di));
609  cf3sq = 0.5*(tsum + sqrt(tdif*tdif + 4.0*asq*(b1*b1+b2*b2)*di));
610 
611 #else /* MHD */
612 
613  /* compute sound speed squared */
614 #ifdef ADIABATIC
615  p = MAX(Gamma_1*(pGrid->U[k][j][i].E - 0.5*pGrid->U[k][j][i].d*qsq),
616  TINY_NUMBER);
617  asq = Gamma*p*di;
618 #else
619  asq = Iso_csound2;
620 #endif /* ADIABATIC */
621  /* compute fast magnetosonic speed squared in each direction */
622  cf1sq = asq;
623  cf2sq = asq;
624  cf3sq = asq;
625 
626 #endif /* MHD */
627 
628  /* compute maximum inverse of dt (corresponding to minimum dt) */
629  if (pGrid->Nx[0] > 1)
630  max_dti = MAX(max_dti,(fabs(v1)+sqrt((double)cf1sq))/pGrid->dx1);
631  if (pGrid->Nx[1] > 1)
632  max_dti = MAX(max_dti,(fabs(v2)+sqrt((double)cf2sq))/pGrid->dx2);
633  if (pGrid->Nx[2] > 1)
634  max_dti = MAX(max_dti,(fabs(v3)+sqrt((double)cf3sq))/pGrid->dx3);
635  }
636  }
637  }
638 
639  /* get timestep. */
640  dt = CourNo/max_dti;
641 
642 #ifdef MPI_PARALLEL
643  err = MPI_Allreduce(&dt, &dt_glob, 1, MP_RL, MPI_MIN, MPI_COMM_WORLD);
644  if(err) ath_error("[compute_dt_hydro]: MPI_Allreduce returned error code %d\n"
645  ,err);
646  dt = dt_glob;
647 #endif /* MPI_PARALLEL */
648  return(dt);
649 }
