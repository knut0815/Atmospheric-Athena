237  int j;
238  long int k;
239  static long int idum2=123456789;
240  static long int iy=0;
241  static long int iv[NTAB];
242  double temp;
243 
244  if (*idum <= 0) { /* Initialize */
245  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
246  else *idum = -(*idum);
247  idum2=(*idum);
248  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
249  k=(*idum)/IQ1;
250  *idum=IA1*(*idum-k*IQ1)-k*IR1;
251  if (*idum < 0) *idum += IM1;
252  if (j < NTAB) iv[j] = *idum;
253  }
254  iy=iv[0];
255  }
256  k=(*idum)/IQ1; /* Start here when not initializing */
257  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
258  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
259  k=idum2/IQ2;
260  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */

262  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
263  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
264  iv[j] = *idum; /* are combined to generate output */

266  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
