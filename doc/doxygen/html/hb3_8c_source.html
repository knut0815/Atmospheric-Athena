345 double ran2(long int *idum)
346 {
347  int j;
348  long int k;
349  static long int idum2=123456789;
350  static long int iy=0;
351  static long int iv[NTAB];
352  double temp;
353 
354  if (*idum <= 0) { /* Initialize */
355  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
356  else *idum = -(*idum);
357  idum2=(*idum);
358  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
359  k=(*idum)/IQ1;
360  *idum=IA1*(*idum-k*IQ1)-k*IR1;
361  if (*idum < 0) *idum += IM1;
362  if (j < NTAB) iv[j] = *idum;
363  }
364  iy=iv[0];
365  }
366  k=(*idum)/IQ1; /* Start here when not initializing */
367  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
368  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
369  k=idum2/IQ2;
370  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
371  if (idum2 < 0) idum2 += IM2;
372  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
373  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
374  iv[j] = *idum; /* are combined to generate output */
375  if (iy < 1) iy += IMM1;
376  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
377  else return temp;
378 }
