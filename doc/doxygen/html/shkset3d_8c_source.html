569  }
570  }
571  }
572  }
573 #endif /* MHD */

576  for(k=0; k<rz; k++){
577  for(j=0; j<ry; j++){

581 #ifdef ADIABATIC

583 #endif
584

586  * conservative variables. */
587

589  for(jsqa=j*scy; jsqa<(j+1)*scy; jsqa++){

595 #ifdef ADIABATIC
596  qa[k][j][i].E += sqa[ksqa][jsqa][isqa].E;
597 #endif
598  }

605  qa[k][j][i].M1 /= (Real)(N*N);
606  qa[k][j][i].M2 /= (Real)(N*N);
607  qa[k][j][i].M3 /= (Real)(N*N);
608 #ifdef ADIABATIC
609  qa[k][j][i].E /= (Real)(N*N);
610 #endif

613 #ifdef MHD

617 #endif /* MHD */

624  */

627  for (j=0; j<=(pGrid->je)+nghost; j++) {
628  for (i=0; i<=(pGrid->ie)+nghost; i++) {
629

631  * the equivalent fluid element on the interval (0 < iy <= ry) and
632  * (0 < iz <= rz). */

635  mjx = jx = (j-(pGrid->js)) + pGrid->Disp[1];
636  mkx = kx = (k-(pGrid->ks)) + pGrid->Disp[2];

688 #endif

692 #ifdef MHD

694  pGrid->B2i[k][j][i] = aB2i[mkx][mjx][mix - qa_min_ix];
695  pGrid->B3i[k][j][i] = aB3i[mkx][mjx][mix - qa_min_ix];
