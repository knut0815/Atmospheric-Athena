617 Real trapzd(Real (*func)(Real), const Real a, const Real b, const int n,
618  const Real s)
619 {
620  Real x,tnm,sum,dx;
621  int it,j;
622 
623  if (n == 1) {
624  return 0.5*(b-a)*(func(a)+func(b));
625  }
626  else {
627  for (it=1,j=1; j<n-1; j++) it <<= 1; /* it = 2^(n-2) */
628  tnm = it;
629  dx = (b-a)/tnm; /* THIS IS THE SPACING OF THE POINTS TO BE ADDED. */
630  x = a + 0.5*dx;
631  for (sum=0.0,j=1; j<=it; j++,x+=dx) sum += func(x);
632  return 0.5*(s+(b-a)*sum/tnm); /* THIS REPLACES s BY ITS REFINED VALUE. */
633  }
634 }

650 Real qsimp(Real (*func)(Real), const Real a, const Real b)
651 {
652  int j;

656  for (j=1; j<=JMAX; j++) {
657  st = trapzd(func,a,b,j,ost);

659  if (j > 5) /* AVOID SPURIOUS EARLY CONVERGENCE. */
660  if (fabs(s-os) < EPS*fabs(os) || (s == 0.0 && os == 0.0)) return s;
661  os=s;
662  ost=st;
663  }

1012 void ludcmp(Real **a, int n, int *indx, Real *d)
1013 {
1014  int i,imax,j,k;
1015  Real big,dum,sum,temp;
1016  Real *rowscale; /* the implicit scaling of each row */
1017 
1018  rowscale = (Real*)calloc_1d_array(n, sizeof(Real));
1019  *d=1.0; /* No row interchanges yet */
1020 
1021  for (i=0;i<n;i++)
1022  { /* Loop over rows to get the implicit scaling information */
1023  big=0.0;
1024  for (j=0;j<n;j++)
1025  if ((temp=fabs(a[i][j])) > big) big=temp;
1026  if (big == 0.0) ath_error("[LUdecomp]:Input matrix is singular!");
1027  rowscale[i]=1.0/big; /* Save the scaling */
1028  }
1029 
1030  for (j=0;j<n;j++) { /* Loop over columns of Crout's method */
1031  /* Calculate the upper block */
1032  for (i=0;i<j;i++) {
1033  sum=a[i][j];
1034  for (k=0;k<i;k++) sum -= a[i][k]*a[k][j];
1035  a[i][j]=sum;
1036  }
1037  /* Calculate the lower block (first step) */
1038  big=0.0;
1039  for (i=j;i<n;i++) {
1040  sum=a[i][j];
1041  for (k=0;k<j;k++)
1042  sum -= a[i][k]*a[k][j];
1043  a[i][j]=sum;
1044  /* search for the largest pivot element */
1045  if ( (dum=rowscale[i]*fabs(sum)) >= big) {
1046  big=dum;
1047  imax=i;
1048  }
1049  }
1050  /* row interchange */
1051  if (j != imax) {
1052  for (k=0;k<n;k++) {
1053  dum=a[imax][k];
1054  a[imax][k]=a[j][k];
1055  a[j][k]=dum;
1056  }
1057  *d = -(*d);
1058  rowscale[imax]=rowscale[j];
1059  }

1061  /* Calculate the lower block (second step) */
1062  if (a[j][j] == 0.0) a[j][j]=TINY_NUMBER;
1063  dum=1.0/(a[j][j]);
1064  for (i=j+1;i<n;i++) a[i][j] *= dum;
1065  }
