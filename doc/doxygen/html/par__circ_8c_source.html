338 static Vector ParticleTroj(Real t)
339 {
340  Vector pos;
341  Real r1,r2,r3,omgt,sinot, cosot;
342 
343  omgt = omg*t;
344  sinot = sin(omgt); cosot = cos(omgt);
345  r1 = r0dn*n1 + sinot*r0cn1 + cosot*(r01-r0dn*n1);
346  r2 = r0dn*n2 + sinot*r0cn2 + cosot*(r02-r0dn*n2);
347  r3 = r0dn*n3 + sinot*r0cn3 + cosot*(r03-r0dn*n3);
348 
349  pos.x1 = x1c+r1; pos.x2 = x2c+r2; pos.x3 = x3c+r3;

408 double ran2(long int *idum)
409 {
410  int j;
411  long int k;
412  static long int idum2=123456789;
413  static long int iy=0;
414  static long int iv[NTAB];
415  double temp;
416 
417  if (*idum <= 0) { /* Initialize */
418  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
419  else *idum = -(*idum);
420  idum2=(*idum);
421  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
422  k=(*idum)/IQ1;
423  *idum=IA1*(*idum-k*IQ1)-k*IR1;
424  if (*idum < 0) *idum += IM1;
425  if (j < NTAB) iv[j] = *idum;
426  }
427  iy=iv[0];
428  }
429  k=(*idum)/IQ1; /* Start here when not initializing */
430  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
431  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
432  k=idum2/IQ2;
433  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
434  if (idum2 < 0) idum2 += IM2;
435  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
436  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
437  iv[j] = *idum; /* are combined to generate output */
438  if (iy < 1) iy += IMM1;
439  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
440  else return temp;
441 }
