197 Real d(const Real x1, const Real x2, const Real x3) {
198  Real x,y,z,x0,y0,r;
199 
200  if (iprob==2) { /* SINE-WAVE, HYDRO ONLY */
201  return rho0*(1.0 + amp*sin(alpha*x2-phi0));
202  }
203  else if (iprob==3) { /* GAUSSIAN PULSE, HYDRO ONLY */
204  x = x1*cos(x2); y = x1*sin(x2); z = x3;
205  x0 = R0*cos(phi0); y0 = R0*sin(phi0);
206  r = sqrt(SQR(x-x0) + SQR(y-y0) + SQR(z-z0));
207  return rho0*(1.0 + exp(-0.5*SQR(3.0*r/rad)));
208  }
209 
210  return rho0;
211 }
