251 int rtbis(double (*pfun)(double), const double x1, const double x2,
252  const double xacc, const int imax, double *prt)
253 {
254  int i;
255  double xn, xm, xp, dx;
256  double fn = (*pfun)(x1), fm, fp = (*pfun)(x2);
257 
258  if(fn < 0.0 && fp > 0.0){
259  xn = x1;
260  xp = x2;
261  }
262  else if(fn > 0.0 && fp < 0.0){
263  xn = x2;
264  xp = x1;
265 
266  fm = fn;
267  fn = fp;
268  fp = fm;
269  }
270  else if(fn == 0.0){ *prt = x1; return 0; }
271  else if(fp == 0.0){ *prt = x2; return 0; }
272  else{
273  ath_perr(-1,"[rtbis]: Root must be bracketed for bisection\n");
274  ath_perr(-1,"[rtbis]: x1=%g, x2=%g, F(x1)=%g, F(x2)=%g\n",
275  x1,x2,fn,fp);
276  return 1; /* Error */
277  }
278 
279  dx = xp - xn;
280 
281  for(i=0; i<imax; i++){ /* Bisection loop */
282  dx *= 0.5;
283  xm = xn + dx;
284  fm = (*pfun)(xm);
285 
286  if(fm < 0.0){ xn = xm; fn = fm; }
287  else { xp = xm; fp = fm; }
288 
289  if(fabs(dx) < xacc || fm == 0.0){
290  *prt = fp < -fn ? xp : xn; /* Return our best value for the root */
291  return 0;
292  }
293  }
294 
295  ath_perr(-1,"[rtbis]: Too many bisections\n");
296 
297  *prt = fp < -fn ? xp : xn; /* Return our best value for the root */
298 
299  return 1;
300 }
